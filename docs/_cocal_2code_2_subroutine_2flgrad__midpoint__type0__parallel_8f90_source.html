12 real(long) :: gr1, gr2, gr3, hrs, hrsinv
13 real(long) :: dfncdr, dfncdth, dfncdphi, drsdth, drsdphi
14 integer, intent(in) :: irf, itf, ipf
15!
16! --- Compute the xy gradient of a function at fixed theta gridpoint.
17! --- The gradient is evaluated at mid points.
18!
19! --- r, phi derivatives. Theta is fixed
20!
21 dfncdr = 0.5d0 &
22 & *(fnc(irf,ipf ) - fnc(irf-1,ipf ) &
23 & + fnc(irf,ipf-1) - fnc(irf-1,ipf-1))*drginv(irf)
24 dfncdphi = 0.5d0 &
25 & *(fnc(irf ,ipf) - fnc(irf ,ipf-1) &
26 & + fnc(irf-1,ipf) - fnc(irf-1,ipf-1))*dphiginv
27
28 drsdphi = (rs(itf,ipf) - rs(itf,ipf-1))*dphiginv
29
30 hrs = 0.5d0*(rs(itf,ipf) + rs(itf,ipf-1))
31
32 hrsinv = 1.0d0/hrs
33!
34! --- To cartesian component.
35!
36 gr1 = dfncdr*hrsinv
37 gr2 = 0.0d0
38 gr3 = dfncdphi*hrginv(irf)*hrsinv*cosecthg(itf) &
39 & - drsdphi*hrsinv*cosecthg(itf)*gr1
40!
41 dfdx = gr1 * sinthg(itf) * hcosphig(ipf) &
42 & + gr2 * costhg(itf) * hcosphig(ipf) &
43 & - gr3 * hsinphig(ipf)
44 dfdy = gr1 * sinthg(itf) * hsinphig(ipf) &
45 & + gr2 * costhg(itf) * hsinphig(ipf) &
46 & + gr3 * hcosphig(ipf)
47 dfdz = gr1 * costhg(itf) &
48 & - gr2 * sinthg(itf)
49!
50end subroutine flgrad_midpoint_type0_parallel
