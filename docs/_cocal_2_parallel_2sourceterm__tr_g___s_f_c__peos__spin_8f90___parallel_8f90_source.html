5subroutine sourceterm_trg_sfc_peos_spin(sou)
6 use phys_constant, only : long, pi
7 use grid_parameter, only : nrf, ntf, npf
8 use def_metric_on_sfc_cf
9 use def_matter, only : emd, vepxf, vepyf, vepzf
10 use def_matter_parameter, only : ome, ber, radi, confpow, velx
11 use def_velocity_rot
12 use def_vector_x, only : vec_xf
13 use def_vector_phi
14!--------LAB----------------
15 use omp_lib
16 use time_global_omp
17!--------LAB----------------
18 implicit none
19 real(long), pointer :: sou(:,:,:)
20 real(long) :: emdfc, rhofc, prefc, hhfc, rp2s, utfc
21 real(long) :: psifc, alpfc, ene
22 real(long) :: bvxdfc, bvydfc, bvzdfc
23 real(long) :: vepxfc, vepyfc, vepzfc, lam
24 real(long) :: wxfc, wyfc, wzfc, psifc4, psifcp, alpfc2, dvep2, wdvep, w2, &
25 & wterm, uih2, hut
26 integer :: irf, itf, ipf
27!----LAB---------------------------------
28 real(long) :: p_time
29 real(long) :: ovdfc1, ovdfc2, ovdfc3
30!----LAB---------------------------------
31!
32! --- Source term of the spatial trace of Einstein eq
33! --- for computing alpha*psi.
34!
35!-------------PARALLEL LAB-------------------------------------------------------------------------
36 p_time = omp_get_wtime()
37!
38!$OMP PARALLEL
39!$OMP DO COLLAPSE(3) PRIVATE(psifc, alpfc, bvxdfc, bvydfc, bvzdfc, emdfc, vepxfc, vepyfc, vepzfc) &
40!$OMP& PRIVATE(wxfc, wyfc, wzfc, hhfc, prefc, rhofc, ene, ovdfc1, ovdfc2, ovdfc3) &
41!$OMP& PRIVATE(lam, psifc4, psifcp, alpfc2, dvep2, wdvep, w2, wterm, uih2, hut, utfc, rp2s)
42 do ipf = 0, npf
43 do itf = 0, ntf
44 do irf = 0, nrf
45 psifc = psif(irf,itf,ipf)
46 alpfc = alphf(irf,itf,ipf)
47 bvxdfc = bvxdf(irf,itf,ipf)
48 bvydfc = bvydf(irf,itf,ipf)
49 bvzdfc = bvzdf(irf,itf,ipf)
50 emdfc = emd(irf,itf,ipf)
51 wxfc = wxspf(irf,itf,ipf)
52 wyfc = wyspf(irf,itf,ipf)
53 wzfc = wzspf(irf,itf,ipf)
54 vepxfc = vepxf(irf,itf,ipf)
55 vepyfc = vepyf(irf,itf,ipf)
56 vepzfc = vepzf(irf,itf,ipf)
57
58 if (irf.eq.nrf) then
59! emdfc = 1.0d-15
60 emdfc = 0.0d0
61 end if
62 call peos_q2hprho(emdfc, hhfc, prefc, rhofc, ene)
63
64 ovdfc1 = bvxdfc + ome*vec_phif(irf,itf,ipf,1) + velx*vec_xf(irf,itf,ipf,1)
65 ovdfc2 = bvydfc + ome*vec_phif(irf,itf,ipf,2) + velx*vec_xf(irf,itf,ipf,2)
66 ovdfc3 = bvzdfc + ome*vec_phif(irf,itf,ipf,3) + velx*vec_xf(irf,itf,ipf,3)
67
68 lam = ber + ovdfc1*vepxfc + ovdfc2*vepyfc + ovdfc3*vepzfc
69 psifc4 = psifc**4
70 psifcp = psifc**confpow
71 alpfc2 = alpfc**2
72
73 dvep2 = (vepxfc**2 + vepyfc**2 + vepzfc**2)/psifc4
74 wdvep = (wxfc*vepxfc + wyfc*vepyfc + wzfc*vepzfc)*psifcp
75 w2 = psifc4*(wxfc*wxfc + wyfc*wyfc + wzfc*wzfc)*psifcp**2.0d0
76
77 wterm = wdvep + w2
78 uih2 = dvep2 + 2.0d0*wdvep + w2
79
80 if ( (lam*lam + 4.0d0*alpfc2*wterm)<0.0d0 ) then
81 write(6,*) "hut imaginary....exiting"
82 stop
83 end if
84 hut = (lam + sqrt(lam*lam + 4.0d0*alpfc2*wterm))/(2.0d0*alpfc2)
85
86 utfc = hut/hhfc
87
88 rp2s = 3.0d0*hhfc*rhofc*(alpfc*utfc)**2 &
89 & - 2.0d0*hhfc*rhofc + 5.0d0*prefc
90!
91 sou(irf,itf,ipf) = radi**2*2.0d0*pi*alpfc*psifc**5*rp2s
92 end do
93 end do
94 end do
95!
96!$OMP END DO
97!$OMP END PARALLEL
98!--------------------------------------------------------------------------------------------------------
99 p_time = omp_get_wtime() - p_time
100 g_time = g_time + p_time
101!
102!------------------------------------- LAB ---------------------------------------------------------------
103!
104end subroutine sourceterm_trg_sfc_peos_spin
