48!
49! -- 3. Drichlet at rgin and rgout

103 gfnsf_sd(il,ir,2) = - 0.0d0
104 gfnsf_sd(il,ir,3) = 0.0d0
105 gfnsf_sd(il,ir,4) = - fac1*fac3*radsq
106!
107 end do
108!$OMP END DO
109!
110
111!
112!$OMP DO COLLAPSE(2) PRIVATE(fac1, fac2, fac3)
113!
