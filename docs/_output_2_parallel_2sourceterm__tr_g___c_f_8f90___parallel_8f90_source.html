4subroutine sourceterm_trg_cf(sou)
5 use phys_constant, only : long
6 use grid_parameter, only : nrg, ntg, npg
7 use def_metric, only : psi, alph, bvxu, bvyu, bvzu, &
8 & tfkijkij, trk
9!!! use def_metric_rotshift, only : ovxu, ovyu, ovzu

23!----LAB---------------------------------

27
28!
29! --- Source term of the spatial trace of Einstein eq
30! --- for computing alpha*psi.
31!
32 fac512 = 5.0d0/12.0d0
33!
34!-------------PARALLEL LAB-------------------------------------------------------------------------
35 p_time = omp_get_wtime()
36!
37!$OMP PARALLEL
38!$OMP DO COLLAPSE(3) PRIVATE( psigc, alpgc, bxugc, byugc, bzugc, dKdx, dKdy, dKdz, aijaij, traceK, LiebeK )
39!
40 do ipg = 1, npg
41 do itg = 1, ntg
42 do irg = 1, nrg
43 call interpo_linear_type0(psigc,psi,irg,itg,ipg)
44 call interpo_linear_type0(alpgc,alph,irg,itg,ipg)
45 call interpo_linear_type0(bxugc,bvxu,irg,itg,ipg)
46 call interpo_linear_type0(byugc,bvyu,irg,itg,ipg)

48!!! call interpo_linear_type0(oxugc,ovxu,irg,itg,ipg)
49!!! call interpo_linear_type0(oyugc,ovyu,irg,itg,ipg)
50!!! call interpo_linear_type0(ozugc,ovzu,irg,itg,ipg)
51
52 call grgrad_midpoint_type0(trk_grid,dkdx,dkdy,dkdz,irg,itg,ipg)
53
54 aijaij = tfkijkij(irg,itg,ipg)
55 tracek = trk(irg,itg,ipg)
56! -- Should be rotating shift for Helical Symmetry.
57 liebek = bxugc*dkdx + byugc*dkdy + bzugc*dkdz
58!!! LieomeK = oxugc*dKdx + oyugc*dKdy + ozugc*dKdz
59!
60 sou(irg,itg,ipg) = alpgc*psigc**5*(0.875d0*aijaij + fac512*tracek**2) &
61 & + psigc**5*liebek
62!!! & + psigc**5*LieomeK
63 end do
64 end do
65 end do
66!
67!$OMP END DO
68!$OMP END PARALLEL
69!----------------------------------------------------------------------------------------------------
70!
71!----------------------------------------------------------------------------------------------------
72 p_time = omp_get_wtime() - p_time
73 g_time = g_time + p_time

76!
