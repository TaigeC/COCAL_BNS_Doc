2!*********************************************************************
3! subroutine rk numerically integrates a system of neq
4! first order ordinary differential equations of the form
5! dy(i)/dx = f(x, y(1), ..., y(neq)),
6! by the classical runge-kutta formula.
7!
8! parameters
9! === input ===
10! (1) neq: number of equations to be integrated
11! (2) func: subroutine func(x, y, f) to evaluate derivatives
12! f(i)=dy(i)/dx
13! (3) x0: initial value of independent variable
14! (4) xe: output point at which the solution is desired
15! (5) n: number of divisions
16! the interval (x0, xe) is divided into n subintervals
17! with the length (xe-x0)/n and in each subinterval
18! the classical runge-kutta formula is used.
19! (6) y0(i) (i=1, .., neq): initial value at x0
20! === output ===
21! (7) yn(i) (i=1, .., neq): approximate solution at xe
22! === other ===
23! (8) work(): two-dimentional array (size=(neq, 2)) to be
24! used inside rk
25! copyright: m. sugihara, november 15, 1989, v. 1
26!*********************************************************************
27 implicit none
28 external func
29 integer, intent(in) :: neq, n
30 real(8), intent(in) :: rfac
31 real(8), intent(inout) :: x0, y0(neq), work(neq, 2)
32 real(8), intent(in) :: xe
33 real(8), intent(out) :: yn(neq)
34 real(8) :: h
35 integer :: i
36 h = (xe - x0) / n
37 do i = 1, n
38 call rkstep_isotr(neq, func, x0, h, y0, yn, work(1, 1), work(1, 2),rfac)
39 x0 = x0 + h
40 y0(1:neq) = yn(1:neq)
41 end do
42 x0 = xe
