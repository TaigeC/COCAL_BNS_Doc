13! --- calculate sine ,cosine and tange on grids of polar angle theta. ---
14! --- notice ; tange diverge to infinity when agr=pi/2
15!
16subroutine trig_grav_theta
17IMPLICIT NONE
18INTEGER :: it
19 DO it = 0, ntg
20 sinthg(it) = sin(thg(it))

22 END DO
23 cosecthg( 0) = 0.0e0 ! should be +inf, should not be used in the code
24 cotanthg( 0) = 0.0e0 ! should be +inf, should not be used in the code
25 DO it = 1, ntg - 1
26 cosecthg(it) = 1.0e0/sin(thg(it))
27 cotanthg(it) = 1.0e0/tan(thg(it))
28 END DO
29 cosecthg(ntg) = 1.0e0 ! should be +inf, should not be used in the code
30 cotanthg(ntg) = 0.0e0 ! should be -inf, should not be used in the code
31!
32 DO it = 1, ntg
33 hsinthg(it) = sin(hthg(it))
34 hcosthg(it) = cos(hthg(it))
35 END DO
36 DO it = 1, ntg
37 hcosecthg(it) = 1.0e0/sin(hthg(it))
