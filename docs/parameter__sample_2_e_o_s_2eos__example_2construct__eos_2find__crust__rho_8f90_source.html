30! -- cgs to g = c = msol = 1 unit.
31! -- assume pre = pre_0 dyn/cm^2 at rho = rho_0 gr/cm^3.
32! -- typically pre_0 = 1.0d+37 dyn/cm^2
33! -- and rho_0 = 1.0d+16 gr/cm^3.
34! -- rescale interface values
35!
36 facrho = (g/c**2)**3*solmas**2

41 end do

44!
45 abc(iphase) = pre_0/rho_0**abi(iphase)
46 abc(iphase) = facpre/facrho**abi(iphase)*abc(iphase)
47 abccgs(iphase) = pre_0/(rho_0**abi(iphase))
48!
49 if (iphase.gt.0) then
50 do ii = iphase-1, 0, -1
51 abc( ii) = rhoi( ii)**(abi(ii+1)-abi(ii))*abc( ii+1)
52 abccgs(ii) = rhocgs(ii)**(abi(ii+1)-abi(ii))*abccgs(ii+1)
53 end do
54 end if
55 if (iphase.lt.nphase) then
56 do ii = iphase+1, nphase
57 abc( ii) = rhoi( ii-1)**(abi(ii-1)-abi(ii))*abc( ii-1)
58 abccgs(ii) = rhocgs(ii-1)**(abi(ii-1)-abi(ii))*abccgs(ii-1)
59 end do
60 end if

64 end do
65!

69 hi(ii) = hi(ii-1) + fac2*(qi(ii) - qi(ii-1))
70 end do

76 & qi(ii), hi(ii), abccgs(ii), rhocgs(ii), &
77 & abccgs(ii)*rhocgs(ii)**abi(ii)
78 end do
79 close(860)
80!
81 rhoini_gcm1 = facrho*rhoini_cgs
82 call peos_lookup(rhoini_gcm1,rhoi,iphase)
83 emdini_gcm1 = abc(iphase)*rhoini_gcm1**(abi(iphase)-1.0d0)
84!

87 k_cru = 3.59389d+13
88!***********************************************************************
89!
90 rho1 = (k_cru/abccgs(0))**(1.0d0/(abi(0)-gamma_cru))
91!
92 write(6,'(a50,1p,e13.5)') "Dividing density between core and crust is rho1 = ", rho1
93 write(6,*) "Add the following two lines to file peos_parameter.dat"
94 write(6,'(1p,2e13.5,a17)') rho1, gamma_cru, " : rho, gamma"
95 write(6,'(1p,2e13.5,a17)') 1.0d0, gamma_cru, " : rho, gamma"
