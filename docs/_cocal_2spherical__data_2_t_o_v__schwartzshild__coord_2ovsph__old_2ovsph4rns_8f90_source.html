251subroutine rk(neq, func, x0, xe, n, y0, yn, work)
252!*********************************************************************
253! subroutine rk numerically integrates a system of neq
254! first order ordinary differential equations of the form
255! dy(i)/dx = f(x, y(1), ..., y(neq)),
256! by the classical runge-kutta formula.
257!
258! parameters
259! === input ===
260! (1) neq: number of equations to be integrated
261! (2) func: subroutine func(x, y, f) to evaluate derivatives
262! f(i)=dy(i)/dx
263! (3) x0: initial value of independent variable
264! (4) xe: output point at which the solution is desired
265! (5) n: number of divisions
266! the interval (x0, xe) is divided into n subintervals
267! with the length (xe-x0)/n and in each subinterval
268! the classical runge-kutta formula is used.
269! (6) y0(i) (i=1, .., neq): initial value at x0
270! === output ===
271! (7) yn(i) (i=1, .., neq): approximate solution at xe
272! === other ===
273! (8) work(): two-dimentional array (size=(neq, 2)) to be
274! used inside rk
275! copyright: m. sugihara, november 15, 1989, v. 1
276!*********************************************************************
277 implicit none
278 external func
279 integer, intent(in) :: neq, n
280 real(8), intent(inout) :: x0, y0(neq), work(neq, 2)
281 real(8), intent(in) :: xe
282 real(8), intent(out) :: yn(neq)
283 real(8) :: h
284 integer :: i
285 h = (xe - x0) / n
286 do i = 1, n
287 call rkstep(neq, func, x0, h, y0, yn, work(1, 1), work(1, 2))
288 x0 = x0 + h
289 y0(1:neq) = yn(1:neq)
290 end do
291 x0 = xe
292 return
293end subroutine rk

316 end do

318 call func(x + h, w, ak)
319 do i = 1, neq

321 end do
