4subroutine calc_vector_phi_grav(sb)
5 use phys_constant, only : long

13!--------ATSOK-LAB----------------

22!----ATSOK-LAB---------------------------------

27!
28!-------------PARALLEL ATSOK-LAB-------------------------------------------------------------------------

30!

32!
33!$OMP PARALLEL

35 do ipg = 0, npg
36 do itg = 0, ntg
37 do irg = 0, nrg
38 rrgg = rg(irg)
39 sth = sinthg(itg)

41 cphi = cosphig(ipg)
42 vec_phig(irg,itg,ipg,1) = - rrgg*sth*sphi
43 vec_phig(irg,itg,ipg,2) = - dis_bin + rrgg*sth*cphi
44 vec_phig(irg,itg,ipg,3) = 0.0d0
45 if (irg.eq.0.or.itg.eq.0.or.ipg.eq.0) cycle
46 rrgg = hrg(irg)
47 sth = hsinthg(itg)
48 sphi = hsinphig(ipg)
49 cphi = hcosphig(ipg)
50 hvec_phig(irg,itg,ipg,1) = - rrgg*sth*sphi
51 hvec_phig(irg,itg,ipg,2) = - dis_bin + rrgg*sth*cphi
52 hvec_phig(irg,itg,ipg,3) = 0.0d0
53 end do
54 end do
55 end do
56!$OMP END DO
57!$OMP END PARALLEL
58!--------------------------------------------------------------------------------------------------------
59 p_time = omp_get_wtime() - p_time

63end subroutine calc_vector_phi_grav
