194function fn_lagint(x,y,v)
195 implicit none
196 real(8), intent(in) :: x(4),y(4), v
197 real(8) :: dx12, dx13, dx14, dx21, dx23, dx24, dx31, dx32, dx34, &
198 & dx41, dx42, dx43, wex1, wex2, wex3, wex4, &
199 & xv1, xv2, xv3, xv4, fn_lagint
200!
201 dx12 = x(1) - x(2)
202 dx13 = x(1) - x(3)
203 dx14 = x(1) - x(4)
204 dx23 = x(2) - x(3)
205 dx24 = x(2) - x(4)
206 dx34 = x(3) - x(4)
207 dx21 = - dx12
208 dx31 = - dx13
209 dx32 = - dx23
210 dx41 = - dx14
211 dx42 = - dx24
212 dx43 = - dx34
213 xv1 = v - x(1)
214 xv2 = v - x(2)
215 xv3 = v - x(3)
216 xv4 = v - x(4)
217 wex1 = xv2*xv3*xv4/(dx12*dx13*dx14)
218 wex2 = xv1*xv3*xv4/(dx21*dx23*dx24)
219 wex3 = xv1*xv2*xv4/(dx31*dx32*dx34)
220 wex4 = xv1*xv2*xv3/(dx41*dx42*dx43)
221!
222 fn_lagint = wex1*y(1) + wex2*y(2) + wex3*y(3) + wex4*y(4)
223!
224end function fn_lagint
