343 & + hhyxu*d2hzz(2,1) + hhyyu*d2hzz(2,2) + hhyzu*d2hzz(2,3) &
344 & + hhzxu*d2hzz(3,1) + hhzyu*d2hzz(3,2) + hhzzu*d2hzz(3,3)
345!
346! --- -1/2 Laplacian(h_ab)
347!
348 delhab(1) = d2hxx(1,1) + d2hxx(2,2) + d2hxx(3,3)
349 delhab(2) = d2hxy(1,1) + d2hxy(2,2) + d2hxy(3,3)
350 delhab(3) = d2hxz(1,1) + d2hxz(2,2) + d2hxz(3,3)
351 delhab(4) = d2hyy(1,1) + d2hyy(2,2) + d2hyy(3,3)
352 delhab(5) = d2hyz(1,1) + d2hyz(2,2) + d2hyz(3,3)
353 delhab(6) = d2hzz(1,1) + d2hzz(2,2) + d2hzz(3,3)
354!
355! --- Ricci tensor R_ab and RNL_ab.
356!
357 rabnl(irg,itg,ipg,1) = - 0.5d0*(dhdh(1) + hd2h(1)) &
358 & - dckkx(1) + clxxckkl - ckxlclxk
359 rabnl(irg,itg,ipg,2) = - 0.5d0*(dhdh(2) + hd2h(2)) &
360 & - dckky(1) + clxyckkl - ckxlclyk
361 rabnl(irg,itg,ipg,3) = - 0.5d0*(dhdh(3) + hd2h(3)) &
362 & - dckkz(1) + clxzckkl - ckxlclzk
363 rabnl(irg,itg,ipg,4) = - 0.5d0*(dhdh(4) + hd2h(4)) &
364 & - dckky(2) + clyyckkl - ckylclyk
365 rabnl(irg,itg,ipg,5) = - 0.5d0*(dhdh(5) + hd2h(5)) &
366 & - dckkz(2) + clyzckkl - ckylclzk
367 rabnl(irg,itg,ipg,6) = - 0.5d0*(dhdh(6) + hd2h(6)) &
368 & - dckkz(3) + clzzckkl - ckzlclzk
369 rab(irg,itg,ipg,1:6) = - 0.5d0*delhab(1:6) + rabnl(irg,itg,ipg,1:6)
370!
371 end do
372 end do
