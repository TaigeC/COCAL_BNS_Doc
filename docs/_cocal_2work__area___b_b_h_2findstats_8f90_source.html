237function fn_lagint(x,y,v)
238 implicit none
239 real(8), intent(in) :: x(4),y(4), v
240 real(8) :: dx12, dx13, dx14, dx21, dx23, dx24, dx31, dx32, dx34, &
241 & dx41, dx42, dx43, wex1, wex2, wex3, wex4, &
242 & xv1, xv2, xv3, xv4, fn_lagint
243!
244 dx12 = x(1) - x(2)
245 dx13 = x(1) - x(3)
246 dx14 = x(1) - x(4)
247 dx23 = x(2) - x(3)
248 dx24 = x(2) - x(4)
249 dx34 = x(3) - x(4)
250 dx21 = - dx12
251 dx31 = - dx13
252 dx32 = - dx23
253 dx41 = - dx14
254 dx42 = - dx24
255 dx43 = - dx34
256 xv1 = v - x(1)
257 xv2 = v - x(2)
258 xv3 = v - x(3)
259 xv4 = v - x(4)
260 wex1 = xv2*xv3*xv4/(dx12*dx13*dx14)
261 wex2 = xv1*xv3*xv4/(dx21*dx23*dx24)
262 wex3 = xv1*xv2*xv4/(dx31*dx32*dx34)
263 wex4 = xv1*xv2*xv3/(dx41*dx42*dx43)
264!
265 fn_lagint = wex1*y(1) + wex2*y(2) + wex3*y(3) + wex4*y(4)
266!
267end function fn_lagint
