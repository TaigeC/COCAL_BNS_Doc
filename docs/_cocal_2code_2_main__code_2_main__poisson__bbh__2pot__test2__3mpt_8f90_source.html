172! -- Read parameters
173 call allocate_grid_parameter_mpt
174 call allocate_grid_parameter_binary_excision_mpt
175 do impt = 1, nmpt
176 call read_parameter_mpt(impt)
177 call copy_grid_parameter_to_mpt(impt)
178 call read_parameter_binary_excision_mpt(impt)
179 call copy_grid_parameter_binary_excision_to_mpt(impt)
180 end do
181!

183 call set_allocate_size_mpt
184!
185 call allocate_coordinate_patch_kit_grav_mpt

197!
198 call allocate_mpatch_all_test
199 call allocate_grid_points_asymptotic_patch_mpt
200 call allocate_grid_points_binary_in_asympto_mpt
201 call allocate_metric_and_matter_bhns_test_mpt
202!
203 do impt = 1, nmpt
204 call copy_grid_parameter_from_mpt(impt)
205 call copy_grid_parameter_binary_excision_from_mpt(impt)
206 call coordinate_patch_kit_grav_mpt
207 call calc_parameter_binary_excision
208 if (impt.ne.nmpt) then
209 call calc_grid_points_binary_excision
210 call calc_grid_points_binary_in_asympto(impt,nmpt)
211 call copy_grid_points_binary_in_asympto_to_mpt(impt)
212 end if
213 call calc_weight_midpoint_binary_excision
214!!! call calc_weight_midpoint_binary_excision_hybrid
215 call calc_vector_x_grav(0)
216 call calc_vector_x_matter(0)
217 call calc_vector_phi_grav(0)
218 call calc_vector_phi_matter(0)
219!!! unnecessary for BBH call test_source_mpt(impt)
220 call copy_to_mpatch_all_test(impt)
221 call copy_metric_and_matter_bhns_test_to_mpt(impt)
222 end do
223 call copy_from_mpatch_all_test(nmpt)
224! -- coordinates of asymptotic patch in central patches
225 do impt = 1, 2
226 call calc_grid_points_asymptotic_patch(impt,nmpt)
227 call copy_grid_points_asymptotic_patch_to_mpt(impt)
228 end do
229! -- Analytic solution for 2 potential problem on all patches.
230 do impt = 1, nmpt
231 call copy_from_mpatch_all_test(impt)
232 call copy_metric_and_matter_bhns_test_from_mpt(impt)
233 call test2_analytic_solution_bhex_mpt(impt) ! alpha > 0 at bh
234 call copy_to_mpatch_all_test(impt)
235 call copy_metric_and_matter_bhns_test_to_mpt(impt)
236 end do
237!
238 call iteration_poisson_bbh_2pot_test2_3mpt(iter_count)
239 if (total_iteration.ge.iter_max) then
240 write(6,*)' ** Solution did not converge **'
241 end if
242! call printout_NS_shape_seq(iseq)
243 do impt = 1, nmpt
244 call copy_from_mpatch_all_test(impt)
245 call copy_metric_and_matter_bhns_test_from_mpt(impt)
246 if (outdata_type.eq.'3D') call io_output_poisson_test_3d_mpt(impt)
247 end do
248!
249END PROGRAM main_poisson_binary_2pot_test_3mpt
