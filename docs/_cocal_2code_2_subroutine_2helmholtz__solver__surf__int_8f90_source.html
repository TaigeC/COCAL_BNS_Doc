1subroutine helmholtz_solver_surf_int(sou_surf,dsou_surf,pot)
2!
3! This subroutine computes the contribution spot(ir,it,ip)
4! to the potential pot from a surface term at the inner or outer surface
5! The integrand is proportional to a function
6! psis = partial_r pot(R,theta,phi)n_l(R) - pot \partial_r n_l(omega R)
7! The routine is adapted from grav4, omitting the r integral and
8! r weighting, but retaining the theta and phi integrals and the
9! corresponding sums
10!
11! P_lm components of the potential at the surface:
12! swork1 is the mth component of pot, weighted by wgdtg
13! spwork1 is the mth component of partial_r pot weighted by wgdtg
14! swork2 is the lm component of pot
15! spwork2 is the lm component of partial_r pot
16!
17 use phys_constant, only : long, pi
18 use grid_parameter, only : nrg, npg, ntg, nlg
19 use coordinate_grav_r, only : rg
20 use radial_green_fn_helmholtz, only : sbsjy, sbsjyp
21 use legendre_fn_grav, only : plmg, hplmg, facnmg, epsig
22 use trigonometry_grav_phi, only : sinmpg, cosmpg, hsinmpg, hcosmpg
23 use weight_midpoint_grav, only : hwtpgsf
24 use make_array_2d
25 use make_array_3d
26 implicit none
27!
28 integer :: im, il
29 integer :: ir, it, itt, ip, ipp
30 real(long) :: pef, wok, wokc, woks, pi4inv, rsqpi4inv
31 real(long), pointer :: sou_surf(:,:),dsou_surf(:,:),pot(:,:,:)
32 real(long), pointer :: work11c(:,:), gwork11c(:,:)
33 real(long), pointer :: work11s(:,:), gwork11s(:,:)
34 real(long), pointer :: work21c(:,:), gwork21c(:,:)
35 real(long), pointer :: work21s(:,:), gwork21s(:,:)
36 real(long), pointer :: work3c(:,:,:)
37 real(long), pointer :: work4c(:,:,:)
38 real(long), pointer :: work3s(:,:,:)
39 real(long), pointer :: work4s(:,:,:)
40!
41 call alloc_array2d( work11c, 1, ntg, 0, nlg)
42 call alloc_array2d(gwork11c, 1, ntg, 0, nlg)
43 call alloc_array2d( work11s, 1, ntg, 0, nlg)
44 call alloc_array2d(gwork11s, 1, ntg, 0, nlg)
45 call alloc_array2d( work21c, 0, nlg, 0, nlg)
46 call alloc_array2d(gwork21c, 0, nlg, 0, nlg)
47 call alloc_array2d( work21s, 0, nlg, 0, nlg)
48 call alloc_array2d(gwork21s, 0, nlg, 0, nlg)
49 call alloc_array3d(work3c, 0, nlg, 0, nlg, 0, nrg)
50 call alloc_array3d(work3s, 0, nlg, 0, nlg, 0, nrg)
51 call alloc_array3d(work4c, 0, nlg, 0, nrg, 0, ntg)
52 call alloc_array3d(work4s, 0, nlg, 0, nrg, 0, ntg)
53!
54! -- integrate on midpoint
55 work11c(:,:) = 0.0d0
56 gwork11c(:,:) = 0.0d0
57 work11s(:,:) = 0.0d0
58 gwork11s(:,:) = 0.0d0
59 do im = 0, nlg
60 do itt = 1, ntg
61 do ipp = 1, npg
62 wokc = hwtpgsf(itt,ipp)*hcosmpg(im,ipp)
63 woks = hwtpgsf(itt,ipp)*hsinmpg(im,ipp)
64 work11c(itt,im) = work11c(itt,im) + wokc*dsou_surf(itt,ipp)
65 gwork11c(itt,im) = gwork11c(itt,im) + wokc*sou_surf(itt,ipp)
66 work11s(itt,im) = work11s(itt,im) + woks*dsou_surf(itt,ipp)
67 gwork11s(itt,im) = gwork11s(itt,im) + woks*sou_surf(itt,ipp)
68 end do
69 end do
70 end do
71!
72 work21c(:,:) = 0.0d0
73 gwork21c(:,:) = 0.0d0
74 work21s(:,:) = 0.0d0
75 gwork21s(:,:) = 0.0d0
76 do il = 0, nlg
77 do im = 0, nlg
78 pef = epsig(im)*facnmg(il,im)
79 do itt = 1, ntg
80 wok = pef*hplmg(il,im,itt)

82 gwork21c(il,im) = gwork21c(il,im) + wok*gwork11c(itt,im)
83 work21s(il,im) = work21s(il,im) + wok*work11s(itt,im)
84 gwork21s(il,im) = gwork21s(il,im) + wok*gwork11s(itt,im)
85 end do
86 end do
87 end do
88!
89! Construct the surface integral for each l,m,r
90! psi = partial_r pot_lm *y_l(m omega r) - pot_lm* partial_r y_l(m omega r)
91! swork3 = psi * (-m omega)(2l+1)
92!
93! sbsjy := j_l(m omega r) y_l(m omega R) (m omega)(2l+1)
94! sbsjyp = j_l(m omega r) y_l'(m omega R)(m omega)^2(2l+1)
95!
96 work3c = 0.0d0
97 work3s = 0.0d0
98 do ir = 0, nrg
99 do il = 0, nlg
100 do im = 0, nlg
101 work3c(il,im,ir) = work21c(il,im)*sbsjy(il,im,ir) &
102 & - gwork21c(il,im)*sbsjyp(il,im,ir)
103 work3s(il,im,ir) = work21s(il,im)*sbsjy(il,im,ir) &
104 & - gwork21s(il,im)*sbsjyp(il,im,ir)
105 end do
106 end do
107 end do
108!
109 work4c = 0.0d0
110 work4s = 0.0d0
111 do it = 0, ntg
112 do ir = 0, nrg
113 do im = 0, nlg
114 do il = 0, nlg
115 work4c(im,ir,it) = work4c(im,ir,it) &
116 & + work3c(il,im,ir)*plmg(il,im,it)
117 work4s(im,ir,it) = work4s(im,ir,it) &
118 & + work3s(il,im,ir)*plmg(il,im,it)
119 end do
120 end do
121 end do
122 end do
123!
124 pot = 0.0d0
125 do ip = 0, npg
126 do it = 0, ntg
127 do ir = 0, nrg
128 do im = 0, nlg
129 pot(ir,it,ip) = pot(ir,it,ip) &
130 & + work4c(im,ir,it)*cosmpg(im,ip) &
131 & + work4s(im,ir,it)*sinmpg(im,ip)
132 end do
133 end do
134 end do
135 end do
136!
137 pi4inv = 1.0d0/4.0d0/pi
138 pot(:,:,:) = pi4inv*pot(:,:,:)
139!
140 deallocate( work11c)
141 deallocate(gwork11c)
142 deallocate( work11s)
143 deallocate(gwork11s)
144 deallocate( work21c)
145 deallocate(gwork21c)
146 deallocate( work21s)
147 deallocate(gwork21s)
148 deallocate(work3c)
149 deallocate(work3s)
150 deallocate(work4c)
151 deallocate(work4s)
152!
153end subroutine helmholtz_solver_surf_int
