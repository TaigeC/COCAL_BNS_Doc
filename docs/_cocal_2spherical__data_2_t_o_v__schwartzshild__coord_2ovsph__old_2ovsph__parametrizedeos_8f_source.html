327 SUBROUTINE rk(NEQ,FUNC,X0,XE,N,Y0,YN,WORK)
328**********************************************************************
329* SUBROUTINE RK NUMERICALLY INTEGRATES A SYSTEM OF NEQ *
330* FIRST ORDER ORDINARY DIFFERENTIAL EQUATIONS OF THE FORM *
331* DY(I)/DX = F(X, Y(1),..., Y(NEQ)), *
332* BY THE CLASSICAL RUNGE-KUTTA FORMULA. *
333* *
334* PARAMETERS *
335* === INPUT === *
336* (1) NEQ: NUMBER OF EQUATIONS TO BE INTEGRATED *
337* (2) FUNC: SUBROUTINE FUNC(X,Y,F) TO EVALUATE DERIVATIVES *
338* F(I)=DY(I)/DX *
339* (3) X0: INITIAL VALUE OF INDEPENDENT VARIABLE *
340* (4) XE: OUTPUT POINT AT WHICH THE SOLUTION IS DESIRED *
341* (5) N: NUMBER OF DIVISIONS *
342* THE INTERVAL (X0, XE) IS DIVIDED INTO N SUBINTERVALS *
343* WITH THE LENGTH (XE-X0)/N AND IN EACH SUBINTERVAL *
344* THE CLASSICAL RUNGE-KUTTA FORMULA IS USED. *
345* (6) Y0(I) (I=1,..,NEQ): INITIAL VALUE AT X0 *
346* === OUTPUT === *
347* (7) YN(I) (I=1,..,NEQ): APPROXIMATE SOLUTION AT XE *
348* === OTHER === *
349* (8) WORK(): TWO-DIMENTIONAL ARRAY (SIZE=(NEQ,2)) TO BE *
350* USED INSIDE RK *
351* COPYRIGHT: M. SUGIHARA, NOVEMBER 15, 1989, V. 1 *
352**********************************************************************
353 IMPLICIT REAL*8(a-h,o-z)
354 EXTERNAL func
355 dimension y0(neq),yn(neq),work(neq,2)
356 h = (xe - x0) / n
357 DO 10 i = 1,n
358 CALL rkstep(neq,func,x0,h,y0,yn,work(1,1),work(1,2))
359 x0 = x0 + h
360 DO 20 j = 1,neq
361 y0(j) = yn(j)
362 20 CONTINUE
363 10 CONTINUE
364 x0 = xe
365 RETURN
366 END
