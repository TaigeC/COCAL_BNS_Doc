48!
49! call alloc_array3d(plmg, 0, nlg, 0, nlg, 0, ntg)

56! call alloc_array3d(hdtyplmg, 0, nlg, 0, nlg, 1, ntg)

58 theta = thg(it)

67 hdtplmg(0:nlg,0:nlg,it) = dtpnag(0:nlg,0:nlg)
68 end do

71!
72! call alloc_array2d(facnmg, 0, nlg, 0, nlg)

76 end do
77 end do

81 fnn = real(nn)
82 facnmg(nn,0) = 1.0d0
83 do mm = 1, nn
84 fmm = real(mm)

111SUBROUTINE legendre_theta(pnag,dtpnag,theta)
112 use phys_constant, only : nnlg, long
113 use grid_parameter, only : nlg
114 implicit none
115 integer :: it, mm, nn, kk
116 real(long) :: fmm, fkk, cc, ss, q1, q2
117 real(long) :: pnag(0:nnlg,0:nnlg), dtpnag(0:nnlg,0:nnlg)
118 real(long) :: theta, fac(0:nnlg)
119!
120 cc = cos(theta)
121 ss = sin(theta)
122!
123 pnag(0:nlg,0:nlg) = 0.0d0
124 dtpnag(0:nlg,0:nlg) = 0.0d0
125!
126 fac(0) = 1.0d0
127 do mm = 1, nlg
128 fmm = real(mm)
129 fac(mm) = (2.0d0*fmm-1.0d0) * fac(mm-1)
130 end do
131!
132 pnag(0,0) = 1.d0
133 dtpnag(0,0) = 0.d0
134 do mm = 1, nlg
135 pnag(mm,mm) = fac(mm) * (-ss)**mm
136 end do
137!
138 mm = 1
139 dtpnag(mm,mm) = fac(mm) * dble(mm)*(-cc)
140 do mm = 2, nlg
141 dtpnag(mm,mm) = fac(mm) * dble(mm)*(-ss)**(mm-1)*(-cc)
142 end do
143!
144 do mm = 0, nlg-1
145 fmm = real(mm)
146 pnag(mm+1,mm) = (2.0d0*fmm + 1.0d0)* cc* pnag(mm,mm)
147 dtpnag(mm+1,mm) = (2.0d0*fmm + 1.0d0)*(cc*dtpnag(mm,mm) &
148 & -ss* pnag(mm,mm))
149 end do
150!
151 do mm = 0, nlg-2
152 fmm = real(mm)
153 do kk = 2, nlg-mm
154 fkk = real(kk)
155 q1 = ( 2.0d0 * fmm + 2.0d0 * fkk - 1.0d0 ) / fkk
156 q2 = ( 2.0d0 * fmm + fkk - 1.0d0 ) / fkk
157 pnag(mm+kk,mm) = q1* cc* pnag(mm+kk-1,mm) - q2* pnag(mm+kk-2,mm)
158 dtpnag(mm+kk,mm) = q1*(cc*dtpnag(mm+kk-1,mm) - ss * pnag(mm+kk-1,mm)) &
159 & - q2*dtpnag(mm+kk-2,mm)
160 end do
161 end do
162end subroutine legendre_theta
