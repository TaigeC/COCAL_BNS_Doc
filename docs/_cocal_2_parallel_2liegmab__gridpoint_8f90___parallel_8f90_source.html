4subroutine liegmab_gridpoint
5 use grid_parameter, only : nrg, ntg, npg
6 use def_lie_derivatives_grid, only : elpxx_grid, elpxy_grid, elpxz_grid, &
7 & elpyy_grid, elpyz_grid, elpzz_grid, &
8 & rlpxx_grid, rlpxy_grid, rlpxz_grid, &
9 & rlpyy_grid, rlpyz_grid, rlpzz_grid

11 & hxxu, hxyu, hxzu, hyyu, hyzu, hzzu
12 use def_dvphi, only : dphiu
13 use interface_grdphi_gridpoint_type0
14!--------LAB----------------
15 use phys_constant, only : long
16 use omp_lib
17 use time_global_omp
18!--------LAB----------------
19 implicit none
20 real(8) :: dhxxdp, dhxxdx, dhxxdy, dhxxdz, &
21 & dhxydp, dhxydx, dhxydy, dhxydz, &
22 & dhxzdp, dhxzdx, dhxzdy, dhxzdz, &
23 & dhyxdp, dhyxdx, dhyxdy, dhyxdz, &
24 & dhyydp, dhyydx, dhyydy, dhyydz, &
25 & dhyzdp, dhyzdx, dhyzdy, dhyzdz, &
26 & dhzxdp, dhzxdx, dhzxdy, dhzxdz, &
27 & dhzydp, dhzydx, dhzydy, dhzydz, &
28 & dhzzdp, dhzzdx, dhzzdy, dhzzdz, &
29 & fa13, trlie, &
30 & gmxxd, gmxyd, gmxzd, gmxxu, gmxyu, gmxzu, &
31 & gmyxd, gmyyd, gmyzd, gmyxu, gmyyu, gmyzu, &
32 & gmzxd, gmzyd, gmzzd, gmzxu, gmzyu, gmzzu
33 integer :: ipg, irg, itg
34!----LAB---------------------------------
35 real(long) :: p_time
36 real(long) :: dphiu_12, dphiu_21
37!----LAB---------------------------------
38
39!
40! --- Compute Lie derivative of tgamma w.r.t. phi, and
41! --- and conformally invariant derivative of tgamma w.r.t. phi.
42! NOTE: dphiu(a,b) = D_b phi^a
43!
44!
45!-------------PARALLEL LAB-------------------------------------------------------------------------
46 p_time = omp_get_wtime()
47!
48 fa13 = 1.0d0/3.0d0
49 dphiu_12 =-1.0d0
50 dphiu_21 = 1.0d0
51!
52!$OMP PARALLEL
53!
54!$OMP DO COLLAPSE(3) PRIVATE( dhxxdp, dhxydp, dhxzdp, dhyxdp, dhyydp, dhyzdp, dhzxdp, dhzydp, dhzzdp ) &
55!$OMP& PRIVATE( gmxxd, gmxxu, gmxyd, gmxyu, gmxzd, gmxzu, gmyxd, gmyxu, gmyyd, gmyyu ) &
56!$OMP& PRIVATE( gmyzd, gmyzu, gmzxd, gmzxu, gmzyd, gmzyu, gmzzd, gmzzu, trlie )
57!
58 do ipg = 0, npg
59 do itg = 0, ntg
60 do irg = 0, nrg
61!
62 gmxxd=hxxd(irg,itg,ipg)
63 gmxyd=hxyd(irg,itg,ipg)
64 gmxzd=hxzd(irg,itg,ipg)
65 gmyyd=hyyd(irg,itg,ipg)
66 gmyzd=hyzd(irg,itg,ipg)
67 gmzzd=hzzd(irg,itg,ipg)
68 gmxxd = gmxxd + 1.0d0
69 gmyyd = gmyyd + 1.0d0
70 gmzzd = gmzzd + 1.0d0
71 gmyxd = gmxyd
72 gmzxd = gmxzd
73 gmzyd = gmyzd
74 gmxxu=hxxu(irg,itg,ipg)
75 gmxyu=hxyu(irg,itg,ipg)
76 gmxzu=hxzu(irg,itg,ipg)
77 gmyyu=hyyu(irg,itg,ipg)
78 gmyzu=hyzu(irg,itg,ipg)
79 gmzzu=hzzu(irg,itg,ipg)
80 gmxxu = gmxxu + 1.0d0
81 gmyyu = gmyyu + 1.0d0
82 gmzzu = gmzzu + 1.0d0
83 gmyxu = gmxyu
84 gmzxu = gmxzu
85 gmzyu = gmyzu
86!
87 call grdphi_gridpoint_type0(hxxd,dhxxdp,irg,itg,ipg)
88 call grdphi_gridpoint_type0(hxyd,dhxydp,irg,itg,ipg)
89 call grdphi_gridpoint_type0(hxzd,dhxzdp,irg,itg,ipg)
90 call grdphi_gridpoint_type0(hyyd,dhyydp,irg,itg,ipg)
91 call grdphi_gridpoint_type0(hyzd,dhyzdp,irg,itg,ipg)
92 call grdphi_gridpoint_type0(hzzd,dhzzdp,irg,itg,ipg)
93!
94 rlpxx_grid(irg,itg,ipg) = dhxxdp + gmxyd*dphiu_21 + gmyxd*dphiu_21
95 rlpxy_grid(irg,itg,ipg) = dhxydp + gmxxd*dphiu_12 + gmyyd*dphiu_21
96 rlpxz_grid(irg,itg,ipg) = dhxzdp + gmyzd*dphiu_21
97 rlpyy_grid(irg,itg,ipg) = dhyydp + gmyxd*dphiu_12 + gmxyd*dphiu_12
98 rlpyz_grid(irg,itg,ipg) = dhyzdp + gmxzd*dphiu_12
99 rlpzz_grid(irg,itg,ipg) = dhzzdp
100!
101 dhxxdp = rlpxx_grid(irg,itg,ipg)
102 dhxydp = rlpxy_grid(irg,itg,ipg)
103 dhxzdp = rlpxz_grid(irg,itg,ipg)
104 dhyydp = rlpyy_grid(irg,itg,ipg)
105 dhyzdp = rlpyz_grid(irg,itg,ipg)
106 dhzzdp = rlpzz_grid(irg,itg,ipg)
107 dhyxdp = dhxydp
108 dhzxdp = dhxzdp
109 dhzydp = dhyzdp
110
111 trlie = gmxxu*dhxxdp + gmxyu*dhxydp + gmxzu*dhxzdp &
112 & + gmyxu*dhyxdp + gmyyu*dhyydp + gmyzu*dhyzdp &
113 & + gmzxu*dhzxdp + gmzyu*dhzydp + gmzzu*dhzzdp
114!
115 elpxx_grid(irg,itg,ipg) = dhxxdp - fa13*gmxxd*trlie
116 elpxy_grid(irg,itg,ipg) = dhxydp - fa13*gmxyd*trlie
117 elpxz_grid(irg,itg,ipg) = dhxzdp - fa13*gmxzd*trlie
118 elpyy_grid(irg,itg,ipg) = dhyydp - fa13*gmyyd*trlie
119 elpyz_grid(irg,itg,ipg) = dhyzdp - fa13*gmyzd*trlie
120 elpzz_grid(irg,itg,ipg) = dhzzdp - fa13*gmzzd*trlie
121!
122 end do
123 end do
124 end do
125!
126!$OMP END DO
127!
128!$OMP END PARALLEL
129!--------------------------------------------------------------------------------------------------------
130!
131!--------------------------------------------------------------------------------------------------------
132 p_time = omp_get_wtime() - p_time
133 g_time = g_time + p_time
134!------------------------------------- LAB ---------------------------------------------------------------
135!
136end subroutine liegmab_gridpoint
