30!
31
32!
33!-------------PARALLEL LAB-------------------------------------------------------------------------
34 p_time = omp_get_wtime()
35!
36! --- For the differential rotation, rotating shift is not used.
37!! if (ROT_LAW.eq.'DR') then
38!!!
39!! ovxu(0:nrg,0:ntg,0:npg) = bvxu(0:nrg,0:ntg,0:npg)

43!! ovyd(0:nrg,0:ntg,0:npg) = bvyd(0:nrg,0:ntg,0:npg)
44!! ovzd(0:nrg,0:ntg,0:npg) = bvzd(0:nrg,0:ntg,0:npg)
45!!!
46!!! --- For the frame rotation
47!! else
48! --- shift to rotating shift.
49!
50
51!
52!$OMP PARALLEL
53!
54!$OMP DO COLLAPSE(3) PRIVATE( vphixg, vphiyg, vphizg, bvxgc, bvygc, bvzgc, omegc )
55!
56 do ipg = 0, npg
57 do itg = 0, ntg
58 do irg = 0, nrg
59 vphixg = vec_phig(irg,itg,ipg,1)
60 vphiyg = vec_phig(irg,itg,ipg,2)

68 ovyu(irg,itg,ipg) = bvygc + omegc*vphiyg
69 ovzu(irg,itg,ipg) = bvzgc + omegc*vphizg
70 end do
71 end do
72 end do
73!
74!$OMP END DO
75
76! --- Lowering index of rotating shift.
77
78!
79!$OMP DO COLLAPSE(3) PRIVATE( gmxxd, gmxyd, gmxzd, gmyxd, gmyyd, gmyzd, gmzxd, gmzyd, gmzzd ) &
80!$OMP& PRIVATE( vphixg, vphiyg, vphizg, bvxgc, bvygc, bvzgc, omegc )
81!
82 do ipg = 0, npg
83 do itg = 0, ntg
84 do irg = 0, nrg
85 gmxxd = 1.0d0 + hxxd(irg,itg,ipg)
86 gmxyd = hxyd(irg,itg,ipg)
87 gmxzd = hxzd(irg,itg,ipg)
88 gmyxd = hxyd(irg,itg,ipg)
89 gmyyd = 1.0d0 + hyyd(irg,itg,ipg)
90 gmyzd = hyzd(irg,itg,ipg)
91 gmzxd = hxzd(irg,itg,ipg)
92 gmzyd = hyzd(irg,itg,ipg)
93 gmzzd = 1.0d0 + hzzd(irg,itg,ipg)
94!
95 vphixg = vec_phig(irg,itg,ipg,1)
96 vphiyg = vec_phig(irg,itg,ipg,2)
97 vphizg = vec_phig(irg,itg,ipg,3)
98 bvxgc = bvxd(irg,itg,ipg)
99 bvygc = bvyd(irg,itg,ipg)
100 bvzgc = bvzd(irg,itg,ipg)
101!! omegc = omeg(irg,itg,ipg)
102 omegc = ome
103!
104 ovxd(irg,itg,ipg) = bvxgc + gmxxd*omegc*vphixg &
105 & + gmxyd*omegc*vphiyg &
106 & + gmxzd*omegc*vphizg
107 ovyd(irg,itg,ipg) = bvygc + gmyxd*omegc*vphixg &
108 & + gmyyd*omegc*vphiyg &
109 & + gmyzd*omegc*vphizg
110 ovzd(irg,itg,ipg) = bvzgc + gmzxd*omegc*vphixg &
111 & + gmzyd*omegc*vphiyg &
112 & + gmzzd*omegc*vphizg
113 end do
114 end do
115 end do
116!
117!$OMP END DO
118!
119!$OMP END PARALLEL
120!--------------------------------------------------------------------------------------------------------
121!
122!! end if
123!
124
125!
126!--------------------------------------------------------------------------------------------------------
127 p_time = omp_get_wtime() - p_time
128 g_time = g_time + p_time
129!------------------------------------- LAB ---------------------------------------------------------------
130!
